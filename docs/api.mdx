```python
# 파이썬 프리티어 지원 안함.. jsExample -> Python 으로

import requests

# URL, 헤더 지정
url = "http://kapi.kakao.com/v1/internal/calendar/create/task"
headers = {
    "Authorization": f"KakaoAK {SERVICE_APP_ADMIN_KEY}",
}

# 데이터 지정
data = {
    "target_id_type": "user_id",
    "target_id": TARGET_ID,
    "task": {
        "content": "테스트 할 일 수정",
        "due_info": {
            "due_date": "20231211",
            "time_zone": "Asia/Seoul",
            "alarm_time": "0900",
            "recur": {
                "rrule": "FREQ=DAILY;",
                "record_on": False
            }
        },
        "bookmark": True
    },
    "task_id": TASK_ID,
}

# 요청
response = requests.post(url, headers=headers, json=data)
```
